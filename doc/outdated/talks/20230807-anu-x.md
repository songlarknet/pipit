Accelerometer driver
```

type response =
 | RspRead data
 | RspError
 | RspOk

type command =
 | CmdNone
 | CmdRead
 | CmdReset

node accelerometer(rsp: stream response): stream command =
  let needs_reset = (rsp = RspError) in
  let cmd =
    if needs_reset
    then CmdReset
    else CmdRead
```
